# Retrieve user changes yaml from running instance
helm get values gitlab -n gitlab -o yaml

# Port forward for a service: Argo CD
0.0.0.0 instead of localhost is important to expose the ports on all interfaces
(to get it to physical host)

kubectl port-forward --address 0.0.0.0 svc/argocd-server -n argocd 8080:443

# Port forward for gitlab in detached mode (type 'fg' in the terminal that ran it to get it back)
kubectl port-forward --address 0.0.0.0 svc/gitlab-webservice-default -n gitlab 8081:8080 > /dev/null 2>&1 &


# Inspect sockets
ss  socket statistics
-t	Show TCP sockets
-u	Show UDP sockets
-l	Show only listening sockets (i.e. those waiting for connections)
-n	Show numeric addresses/ports (do not resolve names like localhost)
ss -tuln

# get pw argocd
kubectl -n argocd get secret argocd-initial-admin-secret -o jsonpath="{.data.password}" | base64 -d

# get pw gitlab
kubectl get secret -n gitlab gitlab-gitlab-initial-root-password -o jsonpath="{.data.password}" | base64 --decode
